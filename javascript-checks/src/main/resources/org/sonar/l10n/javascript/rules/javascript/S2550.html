当使用模型defaults(包含数组或对象)的Backbone.js框架时，defaults应该定义为函数而不是对象。这是因为对象和数组是通过JavaScript中的引用传递的。 一个包含数组或对象的defaults对象会设置每个实例的默认值指向相同的共享对象或数组。

相反，使用一个函数，将为每个实例剥离对象或数组的新副本。

不合规的代码示例

var Person = Backbone.Model.extend({
    defaults: {  // 不合规;Person的每个实例都将共享favoriteColors的相同实例
        favoriteColors: ["blue","purple","raspberry"]
    }
});

合规的解决方案

var Person = Backbone.Model.extend({
    defaults: function() {
      return {
        favoriteColors: ["blue","purple","raspberry"]
      };
    }
});
