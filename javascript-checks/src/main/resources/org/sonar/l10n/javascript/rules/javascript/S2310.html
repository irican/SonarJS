循环计数器不应在循环体中修改。但是，表示逻辑值的其他循环控制变量可能在循环中被修改，例如，一个标志表示已经完成了某些操作，然后在for语句中进行测试。

不合规的代码示例

var names = [ "Jack", "Jim", "", "John" ];
for (var i = 0; i < names.length; i++) {
  if (!names[i]) {
    i = names.length;                                 // 不合规
  } else {
    console.log(names[i]);
  }
}

合规的解决方案

var names = [ "Jack", "Jim", "", "John" ];
for (var name of names) {
  if (!name) {
    break;                                 // 合规
  } else {
    console.log(name);
  }
}
