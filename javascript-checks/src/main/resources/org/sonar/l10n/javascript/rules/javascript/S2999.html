new关键字只能用于定义构造函数的对象。将它与其他任何东西一起使用，您将得到一个TypeError，因为没有一个构造函数用于调用new关键字。

不合格的代码示例

function MyClass() {
  this.foo = 'bar';
}

var someClass = 1;

var obj1 = new someClass;    // 不合规;
var obj2 = new MyClass();    // 如果将jsdoc参数设置为true，则不兼容。兼容considerJSDoc = false

合规的解决方案

/**
 * @constructor
 */
function MyClass() {
  this.foo = 'bar';
}

var someClass = function(){
  this.prop = 1;
}

var obj1 = new someClass;  // 合规
var obj2 = new MyClass();  //不管考虑jsdoc值如何，都是合规的
