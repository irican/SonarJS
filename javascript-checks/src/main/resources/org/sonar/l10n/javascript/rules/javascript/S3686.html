创建新对象实例的构造函数只能用new调用。非构造函数不能。混合使用这两种方法可能会在运行时导致意想不到的结果。

不合规的代码示例

function getNum() {
  return 5;
}

function Num(numeric, alphabetic) {
  this.numeric = numeric;
  this.alphabetic = alphabetic;
}

var myFirstNum = getNum();
var my2ndNum = new getNum();  // 不合规.返回一个空对象，而不是5

var myNumObj1 = new Num();
var myNumObj2 = Num();  // 不合规. 返回undefined，而不是对象
