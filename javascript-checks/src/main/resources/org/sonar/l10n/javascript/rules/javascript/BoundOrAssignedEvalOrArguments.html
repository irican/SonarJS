eval 用于将字符串计算为JavaScript代码， arguments 用于通过索引属性访问函数参数。 因此，不应该绑定或分配eval 和 arguments ，因为这样做会覆盖这两个保留字的原始定义。

而且，使用这两个名称中的任何一个来分配或绑定，都会在JavaScript严格模式代码中生成错误。

不合规的代码示例

eval = 17; // 不合规
arguments++; // 不合规
++eval; // 不合规
var obj = { set p(arguments) { } }; // 不合规
var eval; // 不合规
try { } catch (arguments) { } // 不合规
function x(eval) { } // 不合规
function arguments() { } // 不合规
var y = function eval() { }; // 不合规
var f = new Function("arguments", "return 17;"); // 不合规

function fun() {
  if (arguments.length == 0) { // 合规
    // do something
  }
}

合规的解决方案

result = 17;
args++;
++result;
var obj = { set p(arg) { } };
var result;
try { } catch (args) { }
function x(arg) { }
function args() { }
var y = function fun() { };
var f = new Function("args", "return 17;");

function fun() {
  if (arguments.length == 0) {
    // do something
  }
}

弃用

这条规则是不赞成的;使用{rule:javascript:S2137}
