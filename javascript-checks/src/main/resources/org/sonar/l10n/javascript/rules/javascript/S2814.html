此规则检查声明是否使用已使用的名称。实际上，可以将同一个符号作为变量或函数多次使用，但是这样做可能会使维护人员感到困惑。 此外，这种重新分配可能是错误的，开发人员没有意识到变量的值被新赋值覆盖了。

这个规则也适用于函数参数。

不合规的代码示例

var a = 'foo';
function a() {}   // 不合规
console.log(a);   // 打印 "foo"

function myFunc(arg) {
  var arg = "event"; // 不合规, 参数值丢失
}

fun(); // 打印 "bar"

function fun() {
  console.log("foo");
}

fun(); // 打印 "bar"

function fun() {  // 不合规
  console.log("bar");
}

fun(); // 打印 "bar"

合规的解决方案

var a = 'foo';
function otherName() {}
console.log(a);

function myFunc(arg) {
  var newName = "event";
}

fun(); // 打印 "foo"

function fun() {
  print("foo");
}

fun(); // 打印 "foo"

function printBar() {
  print("bar");
}

printBar(); // 打印 "bar"
