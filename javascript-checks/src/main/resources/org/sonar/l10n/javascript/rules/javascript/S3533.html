在ECMAScript 2015之前，模块管理必须是临时的，或者由Node等第三方库提供。Webpack或RequireJS。幸运的是，ES2015提供了模块管理的语言标准机制，import and export，应该转换旧的用法。

不合规的代码示例

// circle.js
exports.area = function (r) {
  return PI * r * r;
};

// foo.js
define(["./cart", "./horse"], function(cart, horse) {  // 不合规
  // ...
});

// bar.js
const circle = require('./circle.js');  // 不合规

合规的解决方案

// circle.js
let area = function (r) {
  return PI * r * r;
}
export default area;

// foo.js
import cart from "./cart.js";
import horse from "./horse.js";

// bar.js
import circle from "./circle.js"
