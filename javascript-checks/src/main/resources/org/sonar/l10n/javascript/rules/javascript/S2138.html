undefined 是尚未创建的变量和属性的值。使用相同的值重置现有变量，您将失去区分存在但没有值的变量和不存在的变量的能力。相反，应该使用 null ，允许您区分已重置的属性和从未创建的属性。

不合规的代码示例

var myObject = {};

// ...
myObject.fname = undefined;  // 不合规
// ...

if (myObject.lname == undefined) {
  // 尚未创建的属性
}
if (myObject.fname == undefined) {
  // 无法知道myObject.fname的真实状态
}

合规的解决方案

var myObject = {};

// ...
myObject.fname = null;
// ...

if (myObject.lname == undefined) {
  // 尚未创建的属性
}
if (myObject.fname == undefined) {
  // 无法知道myObject.fname的真实状态
}
