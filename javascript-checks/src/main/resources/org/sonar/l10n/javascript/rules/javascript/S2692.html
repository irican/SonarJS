大多数针对字符串或数组的indexOf 调用的检查都将其与-1进行比较，因为0是一个有效的索引。 任何查找值>0的检查都会忽略第一个元素，这可能是一个bug。如果只是检查字符串是否存在，可以考虑使用 includes 。

不合规的代码示例

var color = "blue";
var name = "ishmael";
var number = 123;

var arr = [color, name];

if (arr.indexOf("blue") > 0) { // 不合规
  // ...
}
if (arr[0].indexOf("ish") > 0 { // 不合规
  // ...
}

合规的解决方案

var color = "blue";
var name = "ishmael";
var number = 123;

var arr = [color, name];

if (arr.indexOf("blue") >= 0) {
  // ...
}
if (arr[0].includes("ish")) {
  // ...
}
