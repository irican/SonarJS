在函数作用域声明但只在单个块内使用的变量应该在该块中声明，而在块内声明但在块外使用的变量(可以使用var样式的声明)应该在块外声明。

不合规的代码示例

function doSomething(a, b) {
  var i;  // 不合规; 应该在if块中声明
  if (a > b) {
    i = a;
    console.log(i);
    var x = a - b;  // 不合规; 应该在if块外声明
  }

  if (a > 4) {
   console.log(x);
  }

  return a+b;
}

合规的解决方案

function doSomething(a, b) {
  var x = a - b;

  if (a > b) {
    var i = a;
    console.log(i);
  }

  if (a > 4) {
   console.log(x);
  }

  return a+b;
}
